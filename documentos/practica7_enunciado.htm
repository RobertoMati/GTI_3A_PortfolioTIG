    # Cálculo de la distancia parcial y al origen
def distancia(latA, lonA, latB, lonB):
    lonA_rad = m.radians(lonA)
    lonB_rad = m.radians(lonB)
    latA_rad = m.radians(latA)
    latB_rad = m.radians(latB)
    d = m.degrees(m.acos((m.sin(latA_rad)*m.sin(latB_rad))+(m.cos(latA_rad)*m.cos(latB_rad)*m.cos(lonB_rad-lonA_rad))))    
    distancia = 111.11*d*1000
    return d, distancia

    # Cálculo del acimut. Verificar la posición del punto B
def acimut(lonA, latA, lonB, latB):
    latA_rad = m.radians(latA)
    latB_rad = m.radians(latB)
    d, dist= distancia(latA, lonA, latB, lonB)
    d = m.radians(d)
    angulo = m.degrees(m.acos((m.sin(latB_rad)-(m.sin(latA_rad)*m.cos(d)))/(m.cos(latA_rad)*m.sin(d))))
    #print(angulo)
    if lonA>lonB:
        angulo = 360-angulo
    return angulo

def tiempo(tiempo1,tiempo2):
    tiempo1 = datetime.strptime(tiempo1,'%Y-%m-%dT%H:%M:%SZ')
    tiempo2 = datetime.strptime(tiempo2,'%Y-%m-%dT%H:%M:%SZ')
    segundos = (((tiempo2-tiempo1).total_seconds()))
    return segundos/3600


#Función para que la diferencia de tiempo en la grafica se vea en formato H:M:S
def TiempoGrafica(tiempo1,tiempo2):
    tiempo1 = datetime.strptime(tiempo1,'%Y-%m-%dT%H:%M:%SZ')
    tiempo2 = datetime.strptime(tiempo2,'%Y-%m-%dT%H:%M:%SZ')
    #print("Diferencia: " + str(tiempo2-tiempo1))
    segundos = int((((tiempo2-tiempo1).total_seconds())))
    horas = int(segundos/60/60)
    segundos -= horas*60*60
    minutos=int(segundos/60)
    segundos -=minutos*60
    diferenciaS = f"{horas:02d}:{minutos:02d}:{segundos:02d}"
    return diferenciaS

# Cálculo de la velocidad
def velocidad(distancia,tiempo1,tiempo2):
    tiempoF=tiempo(tiempo1,tiempo2)
    velocidad = (distancia/1000)/tiempoF
    #print("Distancia: " + str(distancia/1000))
    #print("Diferencia: " + str(tiempoF))
    #rint("Velocidad: " + str(velocidad))
    return velocidad
